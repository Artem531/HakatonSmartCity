# for CPU version

Для cpu версии нужно установить anaconda + python 3.5.5
Затем в командной строке ввести след. команды для установки библиотек:

conda install pip

conda install numpy==1.15.1

conda install Pillow==5.3.0

conda install scipy==1.1.0

conda install tensorflow-gpu

pip install wget

conda install seaborn==0.9.0

pip install imutils

conda install opencv

conda install dlib

conda install -c conda-forge scikit-image
